JavaScript:

    Toast.success('Operación exitosa');
    
    // Validar archivo
    const file = input.files[0];
    const validacion = FileValidator.validateImage(file);
    if (!validacion.valid) {
        Toast.error(validacion.error);
    }
    
    // Petición AJAX
    AjaxValidator.get('/api/verificar_sesion')
        .then(data => {
            if (data.activa) {
                Toast.info('Sesión activa');
            }
        });
